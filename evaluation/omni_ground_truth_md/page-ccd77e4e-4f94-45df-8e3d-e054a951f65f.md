For some $ {x}_{i} \notin ( {{\left( {\mu }^{-1}\left( \hat{Y}{\tau }^{-\varepsilon }\right) \right) }^{-1},{\mu }^{-1}( {\hat{Y}{\tau }^{-\varepsilon }}) }) $ , we may as well assume that there exist $ {i}_{1} \in \left\{ {{m}_{1},\cdots ,{m}_{{s}_{1}}}\right\} $ and $ {i}_{2} \in \left\{ {{n}_{1},\cdots ,{n}_{{s}_{2}}}\right\} $ such that $ {x}_{{i}_{1}} > {\mu }^{-1}\left( {\hat{Y}{\tau }^{-\varepsilon }}\right) $and $ {x}_{{i}_{2}} < {( {\mu }^{-1}( \hat{Y}{\tau }^{-\varepsilon })) }^{-1} $, respectively. Let $ {s}_{1} \mathrel{\text{:=}}  \left\{ {{m}_{1},\cdots ,{m}_{{s}_{1}}}\right\} $ and$ {s}_{2} \mathrel{\text{:=}} \left\{ {{n}_{1},\cdots ,{n}_{{s}_{2}}}\right\} $. Then

$$
$$\left\langle  {D{f}_{\tau }\left( x\right) y,y}\right\rangle   + {m}_{0}{\left| D{g}_{\tau }\left( x\right) y\right| }^{2} = \mathop{\sum }\limits_{{i = 1,i \notin  {s}_{1} \cup  {s}_{2}}}^{d}\left( {\frac{\partial f\left( x\right) }{\partial {x}_{i}}{y}_{i}^{2} + {m}_{0}{\left| \mathop{\sum }\limits_{{j = 1}}^{m}\frac{\partial {g}^{j}\left( x\right) }{\partial {x}_{i}}{y}_{i}\right| }^{2}}\right)  + {M}_{1} + {M}_{2}, \tag{D2}$$
$$

where

$$
$${M}_{1} = \mathop{\sum }\limits_{{i = {m}_{1}}}^{{m}_{{s}_{1}}}\left( {\frac{\partial {f}_{\tau }\left( x\right) }{\partial {x}_{i}}{y}_{i}^{2} + {m}_{0}{\left| \mathop{\sum }\limits_{{j = 1}}^{m}\frac{\partial {g}_{\tau }^{j}\left( x\right) }{\partial {x}_{i}}{y}_{i}\right| }^{2}}\right)$$
$$

and

$$
$${M}_{2} = \mathop{\sum }\limits_{{i = {n}_{1}}}^{{n}_{{s}_{2}}}\left( {\frac{\partial {f}_{\tau }\left( x\right) }{\partial {x}_{i}}{y}_{i}^{2} + {m}_{0}{\left| \mathop{\sum }\limits_{{j = 1}}^{m}\frac{\partial {g}_{\tau }^{j}\left( x\right) }{\partial {x}_{i}}{y}_{i}\right| }^{2}}\right) .$$
$$

We obtain the following estimation of $ {M}_{1} $ from (D1) that

$$
$$M_{1} =\sum_{i=m_{1}}^{m_{s_{1}}}\left(\frac{\partial f(\Phi(x_{1}),\cdots,\Phi(x_{d}))}{\partial\Phi(x_{i})}y_{i}^{2}\cdot\frac{1}{1+(x_{i}-\mu^{-1}(\hat{Y}\tau^{-\varepsilon}))^{\bar{\mathcal{v}}}}\right.$$
$$

$$
$$+m_{0}\Big|\sum_{j=1}^{m}\frac{\partial g^{j}(\Phi(x_{1}),\cdots,\Phi(x_{d}))}{\partial\Phi(x_{i})}y_{i}\cdot\frac{1}{1+(x_{i}-\mu^{-1}(\hat{Y}\tau^{-\varepsilon}))^{\bar{v}}}\Big|^{2}\Big)$$
$$

$$
$$=\sum_{i=m_{1}}^{m_{s_{1}}}(\frac{1}{(1+(x_{i}-\mu^{-1}(\hat{Y}\tau^{-\varepsilon}))^{\bar{v}})^{2}}\cdot[\frac{\partial f(\Phi(x_{1}),\cdots,\Phi(x_{d}))}{\partial\Phi(x_{i})}y_{i}^{2}$$
$$

$$
$$+m_{0}\Big|\sum_{j=1}^{m}\frac{\partial g^{j}(\Phi(x_{1}),\cdots,\Phi(x_{d}))}{\partial\Phi(x_{i})}y_{i}\Big|^{2}\Big]+\frac{(x_{i}-\mu^{-1}(\hat{Y}\tau^{-\varepsilon}))^{\bar{v}}}{\left(1+(x_{i}-\mu^{-1}(\hat{Y}\tau^{-\varepsilon}))^{\bar{v}}\right)^{2}}\frac{\partial f(\Phi(x_{1}),\cdots,\Phi(x_{d}))}{\partial\Phi(x_{i})}y_{i}^{2}\Big)$$
$$

$$
$$\leq\sum_{i=m_{1}}^{m_{s_{1}}}\left[\frac{\partial f(\Phi(x_{1}),\cdots,\Phi(x_{d}))}{\partial\Phi(x_{i})}y_{i}^{2}+m_{0}\Big|\sum_{j=1}^{m}\frac{\partial g_{j}(\Phi(x_{1}),\cdots,\Phi(x_{d}))}{\partial\Phi(x_{i})}y_{i}\Big|^{2}\right]$$
$$

$$
$$\leq C\sum_{i=m_1}^{m_{s_1}}y_i^2.  \tag{D3}$$
$$

For $ {M}_{2} $ , we analogously obtain

$$
$${M}_{2} \leq  \mathop{\sum }\limits_{{i = {n}_{1}}}^{{n}_{{s}_{2}}}\left\lbrack  {\frac{\partial f\left( {\Phi \left( {x}_{1}\right) ,\cdots ,\Phi \left( {x}_{d}\right) }\right) }{\partial \Phi \left( {x}_{i}\right) }{y}_{i}^{2} + {m}_{0}{\left| \mathop{\sum }\limits_{{j = 1}}^{m}\frac{\partial {g}^{j}\left( {\Phi \left( {x}_{1}\right) ,\cdots ,\Phi \left( {x}_{d}\right) }\right) }{\partial \Phi \left( {x}_{i}\right) }{y}_{i}\right| }^{2}}\right\rbrack   \leq  C\mathop{\sum }\limits_{{i = {n}_{1}}}^{{n}_{{s}_{2}}}{y}_{i}^{2}. \tag{D4}$$
$$

Substituding (D3) and (D4) into (D2) yields that for sufficiently large $ {m}_{0} > 0 $ and any $ x,y \in {\mathbb{R}}^{d} $ ,

$$\left\langle  {D{f}_{\tau }\left( x\right) y,y}\right\rangle   + {m}_{0}{\left| D{g}_{\tau }\left( x\right) y\right| }^{2} \leq  C{\left| y\right| }^{2}. \tag{D5}$$
