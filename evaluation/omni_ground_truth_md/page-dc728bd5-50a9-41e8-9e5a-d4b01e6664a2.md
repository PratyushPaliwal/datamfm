## X.CEN AND Z. SONG

We claim that

$$
$${\int }_{Q}\left| {g\lambda }\right| \mathop{\prod }\limits_{{i \in  \tau }}{\left( \mathop{\sum }\limits_{{P \in  \tilde{\mathcal{S}},P \subseteq  Q}}{\eta }_{i,P}{\chi }_{P}\left( x\right) \right) }^{{k}_{i} - {t}_{i}} \lesssim  {\int }_{Q}{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 1\right) }}^{{k}_{\tau \left( 1\right) } - {t}_{\tau \left( 1\right) }}\left( {{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 2\right) }}^{{k}_{\tau \left( 2\right) } - {t}_{\tau \left( 2\right) }}\left( {{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 3\right) }}^{{k}_{\tau \left( 3\right) } - {t}_{\tau \left( 3\right) }}\left| {g\lambda }\right| }\right) }\right) .\tag{4.27}$$
$$

Clearly, this follows directly from the definitions that

$$
$$\int_{Q}|g\lambda|\prod_{i\in\tau}\left(\sum_{P\in\tilde{\mathcal{S}},P\subseteq Q}\eta_{i,P}\chi_{P}(x)\right)^{k_{i}-t_{i}}$$
$$

$$
$$\lesssim\int_{Q}A_{\tilde{\mathcal{S}},\eta_{\tau(1)}}^{k_{\tau(1)}-t_{\tau(1)}} ( |g\lambda|\left(\sum_{P\in\tilde{\mathcal{S}},P\subseteq Q}\eta_{\tau(2),P}\chi_{P}(x))^{k_{\tau(2)}-t_{\tau(2)}}\left(\sum_{P\in\tilde{\mathcal{S}},P\subseteq Q}\eta_{\tau(3),P}\chi_{P}(x)\right)^{k_{\tau(3)}-t_{\tau(3)}}\right)$$
$$

$$
$$=\int_{Q}A_{\tilde{\mathcal{S}},\eta_{\tau(1)}}^{k_{\tau(1)}-t_{\tau(1)}-1}(A_{\tilde{\mathcal{S}}}\left(|g\lambda|\left(\sum_{P\in\tilde{\mathcal{S}},P\subseteq Q}\eta_{\tau(2),P}\chi_{P})^{k_{\tau(2)}-t_{\tau(2)}}\left(\sum_{P\in\tilde{\mathcal{S}},P\subseteq Q}\eta_{\tau(3),P}\chi_{P}\right)^{k_{\tau(3)}-t_{\tau(3)}}\right)\eta_{\tau(1)}\right)$$
$$

$$
$$=\int_{Q}A_{\tilde{\mathcal{S}},\eta_{r(1)}}^{k_{\tau(1)}-t_{\tau(1)}-1}\left(\eta_{\tau(1)}\sum_{Q^{\prime}\in\tilde{\mathcal{S}}}\langle|g\lambda|\left(\sum_{P\in\tilde{\mathcal{S}},P\subseteq Q}\eta_{\tau(2),P}\chi_{P}\right)^{k_{\tau(2)}-t_{\tau(2)}}\left(\sum_{P\in\tilde{\mathcal{S}},P\subseteq Q}\eta_{\tau(3),P}\chi_{P}\right)^{k_{\tau(3)}-t_{\tau(3)}} \rangle_{Q^{\prime}}\chi_{Q^{\prime}}\right)$$
$$

$$
$$\overset{(4.22)}{\operatorname*{\lesssim}}\int_{Q}A_{\tilde{S},\eta_{\tau(1)}}^{k_{\tau(1)}-t_{\tau(1)}-1}\left(\left(\sum_{Q^{\prime}\in\tilde{S}}\langle A_{\tilde{S},\eta_{\tau(2)}}^{k_{\tau(2)}-t_{\tau(2)}}(A_{\tilde{S},\eta_{\tau(3)}}^{k_{\tau(3)}-t_{\tau(3)}}|g\lambda|)\rangle_{Q^{\prime}}\chi_{Q^{\prime}}\right)\eta_{\tau(1)}\right)$$
$$

$$
$$=\int_{Q}A_{\tilde{\mathcal{S}},\eta_{\tau(1)}}^{k_{\tau(1)}-t_{\tau(1)}}\left(A_{\tilde{\mathcal{S}},\eta_{\tau(2)}}^{k_{\tau(2)}-t_{\tau(2)}}\left(A_{\tilde{\mathcal{S}},\eta_{\tau(3)}}^{k_{\tau(3)}-t_{\tau(3)}}|g\lambda|\right)\right).$$
$$

Combining above estimate (4.27), we obtain

$$
$$\sum_{Q\in\mathcal{S}}\mu(Q)^{\eta}\left(\int_{Q}|g\lambda|\prod_{i\in\tau}|b_{i}-b_{i,Q}|^{k_{i}-t_{i}}\right)\prod_{i\in\tau}\langle\left|f_{i}(b_{i}-b_{i,Q})^{t_{i}}\right|\rangle_{Q}\prod_{j\in\tau}\langle|f_{j}|\rangle_{Q}$$
$$

$$
$$\lesssim\prod_{i\in\tau}\|b_{i}\|_{\mathrm{BMO}_{\eta_{i}}}^{k_{i}}\int_{Q}A_{\tilde{S},\eta_{\tau(1)}}^{k_{\tau(1)}-t_{\tau(1)}}\left(A_{\tilde{S},\eta_{\tau(2)}}^{k_{\tau(2)}-t_{\tau(2)}}\left(A_{\tilde{\mathcal{S}},\eta_{\tau(3)}}^{k_{\tau(3)}-t_{\tau(3)}}|g\lambda|\right)\right)A_{\eta,\tilde{\mathcal{S}}}(\vec{\mathbf{A}})$$
$$

Since $ {A}_{\widetilde{\mathcal{S}}} $ is self-adjoint, we conclude

$$
$$=\prod_{i\in\tau}\|b_{i}\|_{\mathrm{BMO}_{\eta_{i}}}^{k_{i}}\int_{X}A_{\tilde{\mathcal{S}}}\left(A_{\tilde{\mathcal{S}},\eta_{\tau(1)}}^{k_{\tau(1)}-t_{\tau(1)}-1}\left(\mathcal{A}_{\eta,\tilde{\mathcal{S}},\eta_{\tau(1)}}(\vec{\mathbf{A}})\right)\right)A_{\tilde{\mathcal{S}},\eta_{\tau(2)}}^{k_{\tau(2)}-t_{\tau(2)}}\left(A_{\tilde{\mathcal{S}},\eta_{\tau(3)}}^{k_{\tau(3)}-t_{\tau(3)}}|g\lambda|\right)$$
$$

$$
$$= \mathop{\prod }\limits_{{i \in  \tau }}{\|{b}_{i}\|}_{{\mathrm{{BMO}}}_{{\eta }_{i}}}^{{k}_{i}}{\int }_{X}{A}_{\tilde{\mathcal{S}}}\left( {{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 3\right) }}^{{k}_{\tau \left( 3\right) } - {t}_{\tau \left( 3\right) }}\left( {{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 2\right) }}^{{k}_{\tau \left( 2\right) } - {t}_{\tau \left( 2\right) }}\left( {{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 1\right) }}^{{k}_{\tau \left( 1\right) } - {t}_{\tau \left( 1\right) } - 1}\left( {{\mathcal{A}}_{\eta ,\tilde{\mathcal{S}},{\eta }_{\tau \left( 1\right) }}\left( \vec{\mathbf{A}}\right) }\right) }\right) }\right) }\right) \left| {g\lambda }\right| ,$$
$$

where $ {\mathcal{A}}_{\eta ,\tilde{\mathcal{S}},{\eta }_{\tau \left( 1\right) }}\left( \vec{\mathbf{A}}\right) $ means that $ {\mathcal{A}}_{\eta ,\tilde{\mathcal{S}}}\left( \vec{\mathbf{A}}\right) {\eta }_{\tau \left( 1\right) } $ and

$$
$$\vec{\mathbf{A}} = \left( {{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 1\right) }}^{{t}_{\tau \left( 1\right) }}\left( \left| {f}_{\tau \left( 1\right) }\right| \right) ,{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 2\right) }}^{{t}_{\tau \left( 2\right) }}\left( \left| {f}_{\tau \left( 2\right) }\right| \right) ,{A}_{\tilde{\mathcal{S}},{\eta }_{\tau \left( 3\right) }}^{{t}_{\tau \left( 3\right) }}\left( \left| {f}_{\tau \left( 3\right) }\right| \right) ,{f}_{\tau \left( 4\right) }}\right) .$$
$$
