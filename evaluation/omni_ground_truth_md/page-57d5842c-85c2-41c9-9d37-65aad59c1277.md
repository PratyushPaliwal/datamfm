Definition 3.1. Let $ \left( {{\mathfrak{M}}_{\phi },\rho }\right) $ be a representation of an averaging Lie conformal algebra $ \left( {\mathfrak{L},{\left\lbrack \cdot {}_{\lambda} \cdot \right\rbrack }_{P}}\right) $ . Define a map $ {\xi }^{p} : {C}^{p}\left( {\mathfrak{L},\mathfrak{M}}\right) \rightarrow {C}_{AO}^{p}\left( {\mathfrak{L},\mathfrak{M}}\right) $ by

$$
$${\xi }^{p}{f}_{{\lambda }_{1},{\lambda }_{2},\cdots ,{\lambda }_{p - 1}}\left( {{x}_{1},{x}_{2},\cdots ,{x}_{p}}\right)  = {f}_{{\lambda }_{1},{\lambda }_{2},\cdots ,{\lambda }_{p - 1}}\left( {P{x}_{1},P{x}_{2},\cdots ,P{x}_{p}}\right)  - {\left( \phi  \circ  f\right) }_{{\lambda }_{1},{\lambda }_{2},\cdots ,{\lambda }_{p - 1}}\left( {P{x}_{1},{x}_{2},\cdots ,{x}_{p}}\right) ,\tag{15}$$
$$

for all $ {x}_{1},\cdots ,{x}_{p} \in L $ and $ f \in {C}^{p}\left( {\mathfrak{L},\mathfrak{M}}\right) $ .

Lemma 3.2. For any $ {x}_{1},\cdots ,{x}_{p + 1} \in \mathfrak{L} $ and $ f \in {C}^{p}\left( {\mathfrak{L},\mathfrak{M}}\right) $ , we have

$$
$${\xi }^{p + 1}{\left( {\delta }^{p}\left( f\right) \right) }_{{\lambda }_{1},{\lambda }_{2},\cdots ,{\lambda }_{p}}\left( {{x}_{1},{x}_{2},\cdots ,{x}_{p + 1}}\right)  = {\partial }^{p}{\left( {\xi }^{p}\left( f\right) \right) }_{{\lambda }_{1},{\lambda }_{2},\cdots ,{\lambda }_{p}}\left( {{x}_{1},{x}_{2},\cdots ,{x}_{p + 1}}\right) . \tag{16}$$
$$

Proof. First, consider that

$$
$$\xi^{p+1}(\delta^{p}(f))_{\lambda_{1},\lambda_{2},\cdots,\lambda_{_p}}  (x_{1},x_{2},x_{3},\ldots,x_{p+1})$$
$$

$$
$$-\phi(\sum_{i=2}^{p+1}(-1)^{i+1}[x_{i}\lambda_{i}f_{\lambda_{2},\cdots,\hat{\lambda_{i}},\cdots,\lambda_{p+1}}(Px_{1},x_{2},\cdots\hat{x_{i}},\cdots,x_{p+1})]$$
$$

$$
$$+\sum_{2\leq i<j\leq p+1}(-1)^{i+j}f_{\lambda_{i}+\lambda_{j},\lambda_{1},\cdots,\hat{\lambda_{1}},\cdots,\hat{\lambda_{j}},\cdots,\lambda_{p}}([{x_{i}}_{\lambda_{i}}x_{j}],Px_{1},x_{2},\cdots,\hat{x_{i}},\cdots,\hat{x_{j}},\cdots,x_{p+1})).$$
$$

$$
$$=(\delta^{p}f)_{\lambda_{1},\lambda_{2},\cdots,\lambda_{p}}(Px_{1},Px_{2},\ldots,Px_{p},Px_{p+1})$$
$$

$$
$$-(\phi\circ(\delta^pf))_{\lambda_1,\lambda_2,\cdots,\lambda_p}(Px_1,x_2,\ldots,x_p,x_{p+1})$$
$$

$$
$$=\sum_{i=1}^{p+1}(-1)^{i+1}[(Px_{i})_{\lambda_{i}}f_{\lambda_{2},\cdots,\hat{\lambda_{i}},\cdots,\lambda_{p}}(Px_{1},\cdots\hat{Px_{i}},\cdots,Px_{p+1})]$$
$$

$$
$$+\sum_{1\leq i<j\leq p+1}(-1)^{i+j}f_{\lambda_{i}+\lambda_{j},\lambda_{1},\cdots,\hat{\lambda_{i}},\cdots,\hat{\lambda_{j}},\cdots,\lambda_{p}}([Px_{i\lambda_{i}}Px_{j}],Px_{1},\cdots,\hat{Px_{i}},\cdots,\hat{Px_{j}},\cdots,Px_{p+1})$$
$$

$$
$$-\phi([(Px_1)_{\lambda_1}f_{\lambda_2,\cdots,\hat{\lambda_1},\cdots,\lambda_p}(x_2,\cdots,x_{p+1})]$$
$$

$$
$$+\sum_{1<j\leq p+1}(-1)^{1+j}f_{\lambda_{1}+\lambda_{j},\lambda_{2},\cdots,\hat{\lambda_{j}},\cdots,\lambda_{p}}([Px_{1\lambda_{1}}x_{j}],x_{2},\cdots,\hat{x_{j}},\cdots,x_{p+1}))$$
$$

Next

$$
$$\partial_{AO}^{p}(\xi^{p}(f))_{\lambda_{1},\cdots,\lambda_{p}}(x_{1},x_{2},x_{3},\ldots,x_{p+1})$$
$$

$$
$$=\sum_{i=1}^{p+1}(-1)^{i+1}\rho(P(x_{i}))_{\lambda_{i}}(\xi^{p}f)_{\lambda_{1},\cdots,\hat{\lambda_{i}},\cdots,\lambda_{p}}(x_{1},\cdots,\hat{x_{i}},\cdots,x_{p+1})$$
$$

$$
$$+\sum_{1\leq i<j\leq p+1}(-1)^{i+j}(\xi^{p}f)_{\lambda_{i}+\lambda_{j},\lambda_{1},\cdots,\hat{\lambda_{i}},\cdots,\hat{\lambda_{j}},\cdots,\lambda_{p}}([P(x_{i})_{\lambda_{i}}x_{j}],x_{1},\cdots,\hat{x_{i}},\cdots,\hat{x_{j}},\cdots,x_{p+1})$$
$$

$$
$$=\sum_{i=1}^{p+1}(-1)^{i+1}\rho(P(x_{i}))_{\lambda_{i}}f_{\lambda_{1},\lambda_{2},\cdots,\hat{\lambda_{i}},\cdots,\lambda_{p}}(Px_{1},Px_{2},\cdots,\hat{Px_{i}},\cdots,Px_{p+1})$$
$$

$$
$$-\sum_{i=1}^{p+1}(-1)^{i+1}\rho(P(x_{i}))_{\lambda_{i}}(\phi\circ f)_{\lambda_{1},\lambda_{2},\cdots,\hat{\lambda_{i}},\cdots,\lambda_{p}}(Px_{1},x_{2},\cdots,\hat{x_{i}},\cdots,x_{p+1})$$
$$

$$
$$+\sum_{1\leq i<j\leq p+1}(-1)^{i+j}f_{\lambda_{i}+\lambda_{j},\lambda_{1},\cdots,\hat{\lambda_{i}},\cdots,\hat{\lambda_{j}},\cdots,\lambda_{p}}(P[P(x_{i})_{\lambda_{i}}x_{j}],Px_{1},Px_{2},\cdots\hat{Px_{i}},\cdots\hat{Px_{j}},\cdots,Px_{p})$$
$$

$$
$$-\sum_{1\leq i<j\leq p+1}(-1)^{i+j}(\phi\circ f)_{\lambda_{i}+\lambda_{j},\lambda_{1},\lambda_{2},\cdots,\hat{\lambda_{i}},\cdots,\hat{\lambda_{j}},\cdots,\lambda_{p}}(P[P(x_{i})_{\lambda_{i}}x_{j}],x_{1},x_{2},\cdots\hat{x_i},\cdots\hat{x_j},\cdots,x_{p}).$$
$$

By using Eq.(7), we see that

$$
$${\xi }^{p + 1}{\left( {\delta }^{p}\left( f\right) \right) }_{{\lambda }_{1},{\lambda }_{2},\cdots ,{\lambda }_{p}}\left( {{x}_{1},{x}_{2},\cdots ,{x}_{p + 1}}\right)  = {\partial }^{p}{\left( {\xi }^{p}\left( f\right) \right) }_{{\lambda }_{1},{\lambda }_{2},\cdots ,{\lambda }_{p}}\left( {{x}_{1},{x}_{2},\cdots ,{x}_{p + 1}}\right) . \tag{17}$$
$$

By now we complete the proof.
