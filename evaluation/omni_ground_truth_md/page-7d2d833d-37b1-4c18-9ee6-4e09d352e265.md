Let us now consider

$$

$$
$$\leq\left|\int_{0}^{T}\langle\mathrm{B}(\boldsymbol{u}_{n}(t)),\mathrm{P}_{n}\boldsymbol{v}(t)-\boldsymbol{v}(t)\rangle\mathrm{d}t\right|+\left|\int_{0}^{T}\langle\mathrm{B}(\boldsymbol{u}_{n}(t))-\mathrm{B}(\boldsymbol{u}(t)),\boldsymbol{v}(t)-\boldsymbol{\psi}_{\varepsilon}(t)\rangle\mathrm{d}t\right|$$

$$
$$+\left|\int_0^T\langle\mathrm{B}(\boldsymbol{u}_n(t))-\mathrm{B}(\boldsymbol{u}(t)),\boldsymbol{\psi}_\varepsilon(t)\rangle\mathrm{d}t\right|$$

$$
$$\leq\|\mathbf{P}_n-\mathbf{I}_d\|_{\mathcal{L}(\mathbb{V})}\left(\int_0^T\|\boldsymbol{u}_n(t)\|_{\widetilde{\mathbb{L}}^4}^{\frac{8}{d}}\mathrm{d}t\right)^{\frac{d}{4}}\left(\int_0^T\|\boldsymbol{v}(t)\|_{\mathbb{V}}^{\frac{4}{4-d}}\mathrm{d}t\right)^{\frac{4-d}{4}}$$
$$

$$
$$+C\left[\int_{0}^{T}\left(\|\boldsymbol{u}_{n}(t)\|_{\mathbb{\tilde{L}}^{4}}^{\frac{8}{d}}+\|\boldsymbol{u}(t)\|_{\mathbb{\tilde{L}}^{4}}^{\frac{8}{d}}\right)\mathrm{d}t\right]^{\frac{d}{4}}\left(\int_{0}^{T}\|\boldsymbol{v}(t)-\boldsymbol{\psi}_{\varepsilon}(t)\|_{\mathbb{V}^{}}^{\frac{4}{4-d}}\mathrm{d}t\right)^{\frac{4-d}{4}}$$
$$

$$+\left|\int_{0}^{T}\langle\mathrm{B}(\boldsymbol{u}_{n}(t))-\mathrm{B}(\boldsymbol{u}(t)),\boldsymbol{\psi}_{\varepsilon}(t)\rangle\mathrm{d}t\right|$$

$$

Since an application of Ladyzheskaya's inequality and the energy estimate (3.6) yield

$$
$${\int }_{0}^{T}{\|{\mathbf{u}}_{n}\left( t\right) \|}_{{\widetilde{\mathbb{L}}}^{4}}^{\frac{8}{d}}\mathrm{\;d}t \leq  C\mathop{\sup }\limits_{{t \in  \left\lbrack  {0,T}\right\rbrack  }}{\|{\mathbf{u}}_{n}\left( t\right)\|}_{\mathbb{H}}^{\frac{2\left( {4 - d}\right) }{d}}{\int }_{0}^{T}{\|{\mathbf{u}}_{n}\left( t\right)\| }_{\mathbb{V}}^{2}\mathrm{\;d}t \leq  {\left\{  {\|{\mathbf{u}}_{0}\|}_{\mathbb{H}}^{2} + \frac{1}{\mu }{\int }_{0}^{T}\parallel \boldsymbol{f}\left( t\right) {\parallel }_{{\mathbb{V}}^{\prime }}^{2}\mathrm{\;d}t\right\}  }^{\frac{4}{d}},$$
$$

so that

$$
$$\left|\int_0^T\langle\mathrm{B}_n(\boldsymbol{u}_n(t)),\mathrm{P}_n\boldsymbol{v}(t)\rangle\mathrm{d}t-\int_0^T\langle\mathrm{B}(\boldsymbol{u}(t)),\boldsymbol{v}(t)\rangle\mathrm{d}t\right|$$
$$

$$
$$\leq C\left\{\|\boldsymbol{u}_{0}\|_{\mathbb{H}}^{2}+\frac{1}{\mu}\int_{0}^{T}\|\boldsymbol{f}(t)\|_{\mathbb{V}^{\prime}}^{2}\mathrm{d}t\right\}\varepsilon.$$
$$

Since $ \varepsilon > 0 $ is arbitrary, we deduce that

$$
$$\mathop{\lim }\limits_{{n \rightarrow  \infty }}{\int }_{0}^{T}\left\langle  {{\mathrm{B}}_{n}\left( {{\boldsymbol{u}}_{n}\left( t\right) }\right) ,{\mathrm{P}}_{n}\boldsymbol{v}\left( t\right) }\right\rangle  \mathrm{d}t = {\int }_{0}^{T}\langle \mathrm{\;B}\left( {\boldsymbol{u}\left( t\right) }\right) ,\boldsymbol{v}\left( t\right) \rangle \mathrm{d}t. \tag{3.62}$$
$$

We know from (3.6) that

$$
$${\int }_{0}^{T}{\|{\mathcal{C}}_{n}\left( {\mathbf{u}}_{n}\left( t\right) \right) \|}_{\widetilde{\mathbb{L}}^{\frac{r + 1}{r}}}^{\frac{r + 1}{r}}{dt} \leq  {\int }_{0}^{T}{\|{\mathbf{u}}_{n}\left( t\right) \|}_{\widetilde{\mathbb{L}}^{r + 1}}^{r + 1}\mathrm{\;d}t \leq  {\|{\mathbf{u}}_{0}\|}_{\mathbb{H}}^{2} + \frac{1}{\mu }{\int }_{0}^{T}\parallel \boldsymbol{f}\left( t\right) {\parallel }_{{\mathbb{V}}^{\prime }}^{2}\mathrm{\;d}t. \tag{3.63}$$
